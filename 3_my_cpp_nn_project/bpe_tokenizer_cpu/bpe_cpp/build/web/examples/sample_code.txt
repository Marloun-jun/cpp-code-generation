// C++ code example for benchmarks
#include <iostream>
#include <vector>
#include <map>
#include <string>
#include <algorithm>
#include <memory>

// Demo class
class MyClass {
private:
    std::string name_;
    int value_;
    
public:
    MyClass(const std::string& name, int value) 
        : name_(name), value_(value) {}
    
    void print() const {
        std::cout << name_ << ": " << value_ << std::endl;
    }
    
    int getValue() const { return value_; }
    const std::string& getName() const { return name_; }
};

// Template function
template<typename T>
T max(T a, T b) {
    return (a > b) ? a : b;
}

// Main function
int main() {
    // Create vector
    std::vector<int> numbers;
    for (int i = 0; i < 100; ++i) {
        numbers.push_back(i * i);
    }
    
    // Create map
    std::map<std::string, int> scores;
    scores["Alice"] = 95;
    scores["Bob"] = 87;
    scores["Charlie"] = 92;
    
    // Use algorithms
    auto it = std::find(numbers.begin(), numbers.end(), 2500);
    if (it != numbers.end()) {
        std::cout << "Found: " << *it << std::endl;
    }
    
    // Sort
    std::sort(numbers.begin(), numbers.end(), std::greater<int>());
    
    // Use smart pointers
    auto ptr = std::make_unique<MyClass>("Test", 42);
    ptr->print();
    
    // Lambda expressions
    auto sum = [](int a, int b) -> int { return a + b; };
    std::cout << "Sum: " << sum(10, 20) << std::endl;
    
    return 0;
}
